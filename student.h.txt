Student.h
#pragma once

#include <string>

const int MAX_SUBJECTS = 5;

struct Student {
    std::string name;
    std::string subjects[MAX_SUBJECTS];
    int grades[MAX_SUBJECTS];
};

main.cpp
#include <iostream>
#include <string>
#include "student.h"

int main() {
    Student students[3];
    for (int i = 0; i < 3; ++i) {
        std::cout << "Введите данные для студента " << i + 1 << ":" << std::endl;
        std::cout << "Имя: ";
        std::getline(std::cin >> std::ws, students[i].name);
        for (int j = 0; j < MAX_SUBJECTS; ++j) {
            std::cout << "Предмет " << j + 1 << ": ";
            std::getline(std::cin >> std::ws, students[i].subjects[j]);
            std::cout << "Оценка: ";
            std::cin >> students[i].grades[j];
        }
        std::cin.ignore(); // Очищаем буфер после ввода чисел
    }
    for (int i = 0; i < 3; ++i) {
        std::cout << "\nДанные для студента: " << students[i].name << std::endl;
        for (int j = 0; j < MAX_SUBJECTS; ++j) {
            std::cout << students[i].subjects[j] << ": " << students[i].grades[j] << std::endl;
        }
        double sum = 0;
        for (int j = 0; j < MAX_SUBJECTS; ++j) {
            sum += students[i].grades[j];
        }
        double averageGrade = sum / MAX_SUBJECTS;
        std::cout << "Средняя оценка: " << averageGrade << std::endl;
    }

    std::cout << "\nСтуденты со средней оценкой выше 4.0:\n";
    for (int i = 0; i < 3; ++i) {
        double sum = 0;
        for (int j = 0; j < MAX_SUBJECTS; ++j) {
            sum += students[i].grades[j];
        }
        double averageGrade = sum / MAX_SUBJECTS;
        if (averageGrade > 4.0) {
            std::cout << students[i].name << std::endl;
        }
    }

    return 0;
}
